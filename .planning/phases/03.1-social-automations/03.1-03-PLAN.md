```typescript
import { z } from "zod";

export const triggerConfigSchema = z.object({
  keywords: z.array(z.string().min(1).max(100)).min(1).max(50),
  matchMode: z.enum(["exact", "contains"]).default("contains"),
  caseSensitive: z.boolean().default(false),
});

  posts: z.enum(["all", "specific"]).default("all"),
  postIds: z.array(z.string()).optional(),
});

export const actionConfigSchema = z.object({
  messages: z.array(z.string().min(1).max(2000)).min(1).max(20),
});

export const actionSchema = z.object({
  type: z.enum(["reply_comment", "send_dm"]),
  config: actionConfigSchema,
  order: z.number().int().min(0).default(0),
  isActive: z.boolean().default(true),
});

export const automationSchema = z.object({
  name: z.string().min(1).max(100),
  socialAccountId: z.string().uuid(),
  platform: z.enum(["instagram", "tiktok", "youtube", "x"]),
  triggerType: z.enum(["comment_keyword"]).default("comment_keyword"),
  triggerConfig: triggerConfigSchema,
  scope: automationScopeSchema,
  isActive: z.boolean().default(true),
});

export const updateAutomationSchema = automationSchema.partial();

export type AutomationInput = z.infer<typeof automationSchema>;
export type ActionInput = z.infer<typeof actionSchema>;
export type TriggerConfigInput = z.infer<typeof triggerConfigSchema>;
```

```typescript
import { NextRequest, NextResponse } from "next/server";
import { db } from "@/db";
import { automations, automationActions } from "@/db/schema";
import { eq, and, desc } from "drizzle-orm";
import { withAuth } from "@/lib/api-guard";
import { success, error, internalError } from "@/lib/api-response";
import { automationSchema } from "@/lib/automation/validation";
import { z } from "zod";

// GET /api/automations — list user's automations
export async function GET(req: NextRequest) {
  return withAuth(req, async (_req, user) => {
    const list = await db
      .select()
      .from(automations)
      .where(eq(automations.userId, user.id))
      .orderBy(desc(automations.createdAt));

    return success({ automations: list });
  });
}

// POST /api/automations — create automation
export async function POST(req: NextRequest) {
  return withAuth(req, async (_req, user) => {
    try {
      const body = await req.json();
      const parsed = automationSchema.safeParse(body);
      if (!parsed.success) {
        return error("VALIDATION_ERROR", parsed.error.message, 400);
      }

      const data = parsed.data;

      // Verify social account belongs to user
      const { socialAccounts } = await import("@/db/schema");
      const account = await db
        .select()
        .from(socialAccounts)
        .where(and(
          eq(socialAccounts.id, data.socialAccountId),
          eq(socialAccounts.userId, user.id)
        ))
        .limit(1);

      if (!account[0]) {
        return error("NOT_FOUND", "Social account not found", 404);
      }

      const result = await db.insert(automations).values({
        userId: user.id,
        socialAccountId: data.socialAccountId,
        name: data.name,
        platform: data.platform,
        triggerType: data.triggerType,
        triggerConfig: data.triggerConfig,
        scope: data.scope,
        isActive: data.isActive,
      }).returning();

      return success({ automation: result[0] }, 201);
    } catch (err) {
      return internalError(err instanceof Error ? err.message : "Failed to create automation");
    }
  });
}
```

```typescript
import { NextRequest } from "next/server";
import { db } from "@/db";
import { automations, automationActions } from "@/db/schema";
import { eq, and } from "drizzle-orm";
import { withAuth } from "@/lib/api-guard";
import { success, error, internalError, notFound } from "@/lib/api-response";
import { updateAutomationSchema } from "@/lib/automation/validation";

async function getAutomation(id: string, userId: string) {
  const result = await db
    .select()
    .from(automations)
    .where(and(eq(automations.id, id), eq(automations.userId, userId)))
    .limit(1);
  return result[0] || null;
}

// GET /api/automations/[id]
export async function GET(req: NextRequest, { params }: { params: Promise<{ id: string }> }) {
  return withAuth(req, async (_req, user) => {
    const { id } = await params;
    const automation = await getAutomation(id, user.id);
    if (!automation) return notFound("Automation not found");

    const actions = await db
      .select()
      .from(automationActions)
      .where(eq(automationActions.automationId, id))
      .orderBy(automationActions.order);

    return success({ automation, actions });
  });
}

// PUT /api/automations/[id]
export async function PUT(req: NextRequest, { params }: { params: Promise<{ id: string }> }) {
  return withAuth(req, async (_req, user) => {
    const { id } = await params;
    const automation = await getAutomation(id, user.id);
    if (!automation) return notFound("Automation not found");

    try {
      const body = await req.json();
      const parsed = updateAutomationSchema.safeParse(body);
      if (!parsed.success) {
        return error("VALIDATION_ERROR", parsed.error.message, 400);
      }

      const updateData: Record<string, unknown> = {};
      if (parsed.data.name !== undefined) updateData.name = parsed.data.name;
      if (parsed.data.triggerConfig !== undefined) updateData.triggerConfig = parsed.data.triggerConfig;
      if (parsed.data.scope !== undefined) updateData.scope = parsed.data.scope;
      if (parsed.data.isActive !== undefined) updateData.isActive = parsed.data.isActive;
      updateData.updatedAt = new Date();

      const result = await db.update(automations).set(updateData).where(eq(automations.id, id)).returning();
      return success({ automation: result[0] });
    } catch (err) {
      return internalError(err instanceof Error ? err.message : "Failed to update automation");
    }
  });
}

// DELETE /api/automations/[id]
export async function DELETE(req: NextRequest, { params }: { params: Promise<{ id: string }> }) {
  return withAuth(req, async (_req, user) => {
    const { id } = await params;
    const automation = await getAutomation(id, user.id);
    if (!automation) return notFound("Automation not found");

    await db.delete(automations).where(eq(automations.id, id));
    return success({ deleted: true });
  });
}
```

```typescript
async function verifyAutomationOwnership(automationId: string, userId: string) {

// GET /api/automations/[id]/actions

  return withAuth(req, async (_req, user) => {
    const { id } = await params;
    const automation = await verifyAutomationOwnership(id, user.id);
    if (!automation) return notFound("Automation not found");

    try {
      const body = await req.json();

    }

// PUT /api/automations/[id]/actions
export async function PUT(req: NextRequest, { params }: { params: Promise<{ id: string }> }) {
  return withAuth(req, async (_req, user) => {
    const { id } = await params;
    const automation = await verifyAutomationOwnership(id, user.id);
    if (!automation) return notFound("Automation not found");

    try {
      const body = await req.json();
      const { actionId, ...updateData } = body;
      if (!actionId) return error("VALIDATION_ERROR", "actionId is required", 400);

      const result = await db
        .update(automationActions)
        .set({
          ...updateData,
          updatedAt: new Date(),

        .returning();

      return success({ action: result[0] });

      return internalError(err instanceof Error ? err.message : "Failed to update action");
    }
  });
}

    await db
      .delete(automationActions)
      .where(and(eq(automationActions.id, actionId), eq(automationActions.automationId, id)));

    return success({ deleted: true });
  });
}
```

```typescript
      .limit(limit)
      .offset(offset);

  });
```
