```typescript
import { db } from "@/db";
import { socialAccounts } from "@/db/schema";
import { eq } from "drizzle-orm";
import { decrypt } from "@/lib/encryption";
import type { AutomationPlatform } from "../types";

const GRAPH_API_BASE = "https://graph.instagram.com";
const API_VERSION = "v22.0";

export interface PlatformComment {
  id: string;
  text: string;
  username: string;
  timestamp: string;
  userId?: string; // Instagram user ID of commenter
}

export interface InstagramAutomationAdapter {
  platform: AutomationPlatform;
  fetchComments(socialAccountId: string, postId: string, since?: Date): Promise<PlatformComment[]>;
  replyToComment(socialAccountId: string, commentId: string, message: string): Promise<{ success: boolean; error?: string }>;
  sendDM(socialAccountId: string, userId: string, message: string): Promise<{ success: boolean; error?: string }>;
}

export const instagramAdapter: InstagramAutomationAdapter = {
  platform: "instagram",

  async fetchComments(socialAccountId: string, postId: string, since?: Date): Promise<PlatformComment[]> {
    const account = await db.select().from(socialAccounts).where(eq(socialAccounts.id, socialAccountId)).limit(1);
    if (!account[0]) throw new Error("Social account not found");
    if (!account[0].accessToken) throw new Error("No access token");

    const accessToken = await decrypt(account[0].accessToken);
    const url = new URL(`${GRAPH_API_BASE}/${API_VERSION}/${postId}/comments`);
    url.searchParams.set("fields", "id,text,username,timestamp");
    url.searchParams.set("access_token", accessToken);

    const res = await fetch(url.toString());
    const data = await res.json();

    if (data.error) {
      throw new Error(`Failed to fetch comments: ${data.error.message}`);
    }

    const comments: PlatformComment[] = (data.data || []).map((c: Record<string, string>) => ({
      id: c.id,
      text: c.text,
      username: c.username,
      timestamp: c.timestamp,
    }));

    if (since) {
      return comments.filter(c => new Date(c.timestamp) > since);
    }
    return comments;
  },

  async replyToComment(socialAccountId: string, commentId: string, message: string): Promise<{ success: boolean; error?: string }> {
    try {
      const account = await db.select().from(socialAccounts).where(eq(socialAccounts.id, socialAccountId)).limit(1);
      if (!account[0]?.accessToken) return { success: false, error: "No access token" };

      const accessToken = await decrypt(account[0].accessToken);
      const url = new URL(`${GRAPH_API_BASE}/${API_VERSION}/${commentId}/replies`);
      url.searchParams.set("message", message);
      url.searchParams.set("access_token", accessToken);

      const res = await fetch(url.toString(), { method: "POST" });
      const data = await res.json();

      if (data.error) {
        return { success: false, error: data.error.message };
      }

      return { success: true };
    } catch (err) {
      return { success: false, error: err instanceof Error ? err.message : "Unknown error" };
    }
  },

  async sendDM(socialAccountId: string, _userId: string, _message: string): Promise<{ success: boolean; error?: string }> {
    // Instagram Graph API does not support sending DMs directly
    // This requires the Instagram Messaging API which needs app review
    console.log(`[instagram] DM not implemented for account ${socialAccountId} — requires Messaging API permissions`);
    return { success: false, error: "DM_SENDING_NOT_AVAILABLE" };
  },
};
```

Create src/app/api/cron/automation-comments/route.ts:

```typescript
import { NextRequest, NextResponse } from "next/server";
import { db } from "@/db";
import { automations, automationActions, automationLogs, commentWatchState, socialAccounts, platformPosts } from "@/db/schema";
import { eq, and, desc, gte } from "drizzle-orm";
import { matchCommentToAutomations, selectRandomResponse, logAutomationExecution, checkCooldown } from "@/lib/automation/engine";
import { getAdapter } from "@/lib/automation/adapters";
import type { TriggerEvent, AutomationActionResult } from "@/lib/automation/types";

// Vercel Cron will call this endpoint
export async function GET(req: NextRequest) {
  // Verify cron secret to prevent unauthorized access
  const authHeader = req.headers.get("authorization");
  if (authHeader !== `Bearer ${process.env.CRON_SECRET}`) {
    return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
  }

  const results = {
    accountsChecked: 0,
    postsChecked: 0,
    commentsFound: 0,
    automationsTriggered: 0,
    actionsExecuted: 0,
    errors: [] as string[],
  };

  try {
    // Find all active automations grouped by social account
    const activeAutomations = await db
      .select()
      .from(automations)
      .where(eq(automations.isActive, true));

    // Group by social account
    const accountIds = [...new Set(activeAutomations.map(a => a.socialAccountId))];

    for (const socialAccountId of accountIds) {
      results.accountsChecked++;
      const account = await db.select().from(socialAccounts).where(eq(socialAccounts.id, socialAccountId)).limit(1);
      if (!account[0] || !account[0].isActive) continue;

      const platform = account[0].platform;
      const adapter = getAdapter(platform as "instagram" | "tiktok" | "youtube" | "x");
      if (!adapter) {
        results.errors.push(`No adapter for platform: ${platform}`);
        continue;
      }

      // Get automations for this account
      const accountAutomations = activeAutomations.filter(a => a.socialAccountId === socialAccountId);

      // Determine which posts to check
      const postIdsToCheck = new Set<string>();
      for (const automation of accountAutomations) {
        const scope = automation.scope as { posts: "all" | "specific"; postIds?: string[] };
        if (scope.posts === "all") {
          // Fetch recent posts from platform_posts for this account
          const recentPosts = await db
            .select()
            .from(platformPosts)
            .where(eq(platformPosts.platform, platform))
            .orderBy(desc(platformPosts.createdAt))
            .limit(20);
          recentPosts.forEach(p => postIdsToCheck.add(p.externalId));
        } else if (scope.postIds) {
          scope.postIds.forEach(id => postIdsToCheck.add(id));
        }
      }

      for (const postId of postIdsToCheck) {
        results.postsChecked++;

        // Get watch state for this post
        let watchState = await db
          .select()
          .from(commentWatchState)
          .where(and(
            eq(commentWatchState.socialAccountId, socialAccountId),
            eq(commentWatchState.postId, postId)
          ))
          .limit(1);

        const since = watchState[0]?.lastCheckedAt;

        try {
          const comments = await adapter.fetchComments(socialAccountId, postId, since);
          results.commentsFound += comments.length;

          for (const comment of comments) {
            // Find matching automations
            const matches = await matchCommentToAutomations(socialAccountId, comment.text, postId);

            for (const { automation, actions } of matches) {
              results.automationsTriggered++;

              // Check cooldown per commenter
              const cooldownActive = await checkCooldown(automation.id, comment.username, 60);
              if (cooldownActive) continue;

              const actionResults: AutomationActionResult[] = [];

              for (const action of actions) {
                const config = action.config as { messages: string[] };
                const message = selectRandomResponse(config.messages);

                let result: { success: boolean; error?: string };

                if (action.type === "reply_comment") {
                  result = await adapter.replyToComment(socialAccountId, comment.id, message);
                } else if (action.type === "send_dm") {
                  result = await adapter.sendDM(socialAccountId, comment.userId || comment.username, message);
                } else {
                  result = { success: false, error: "Unknown action type" };
                }

                actionResults.push({
                  actionId: action.id,
                  status: result.success ? "success" : "failed",
                  output: result.success ? undefined : result.error,
                  error: result.success ? undefined : result.error,
                });

                if (result.success) results.actionsExecuted++;
              }

              // Log execution
              const triggerEvent: TriggerEvent = {
                type: "comment.received",
                payload: {
                  commentId: comment.id,
                  text: comment.text,
                  username: comment.username,
                  postId,
                  platform,
                },
              };

              await logAutomationExecution(automation.id, triggerEvent, actionResults);
            }
          }

          // Update watch state
          if (watchState[0]) {
            await db.update(commentWatchState)
              .set({ lastCheckedAt: new Date(), lastCommentId: comments[0]?.id })
              .where(eq(commentWatchState.id, watchState[0].id));
          } else {
            await db.insert(commentWatchState).values({
              socialAccountId,
              postId,
              lastCheckedAt: new Date(),
              lastCommentId: comments[0]?.id,
            });
          }
        } catch (err) {
          const msg = err instanceof Error ? err.message : "Unknown error";
          results.errors.push(`Post ${postId}: ${msg}`);
        }
      }
    }

    return NextResponse.json({ success: true, results });
  } catch (err) {
    const message = err instanceof Error ? err.message : "Unknown error";
    return NextResponse.json({ success: false, error: message, results }, { status: 500 });
  }
}
```
